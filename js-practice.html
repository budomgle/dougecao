1. 变量的定义

JavaScript 提供三种声明变量的关键字：var、let、const。

// var：函数作用域，可重复声明（旧写法）
var studentName = "张三";

// let：块级作用域，可重新赋值（推荐）
let age = 20;
age = 21;   // 允许修改

// const：块级作用域，不可重新赋值（常量）
const PI = 3.14159;

2. 变量类型

JavaScript 有 7 种基本数据类型（原始类型）及引用类型。

let str    = "Hello, World!";   // String  字符串
let num    = 42;                  // Number  数字
let bool   = true;               // Boolean 布尔
let undef;                        // Undefined 未定义
let nul    = null;               // Null    空
let sym    = Symbol("id");       // Symbol  唯一标识
let big    = 9007199254740991n;  // BigInt  大整数
let obj    = { name: "李四", age: 18 }; // Object 对象
let arr    = [1, 2, 3];          // Array   数组（特殊对象）

console.log(typeof str);   // "string"
console.log(typeof num);   // "number"
console.log(typeof bool);  // "boolean"

3. 表达式

表达式是能够产生值的代码片段，包含算术、字符串、比较、逻辑、赋值等运算符。

// 算术表达式
let a = 10, b = 3;
console.log(a + b);   // 13
console.log(a - b);   // 7
console.log(a * b);   // 30
console.log(a / b);   // 3.333...
console.log(a % b);   // 1（取余）
console.log(a ** b);  // 1000（幂运算）

// 比较表达式
console.log(a > b);   // true
console.log(a === b);  // false（严格相等）

// 逻辑表达式
console.log(true && false); // false
console.log(true || false);  // true
console.log(!true);          // false

// 字符串拼接
let greeting = "你好，" + "世界！";
// 模板字面量（推荐）
let name = "同学";
let msg = `欢迎你，${name}！`;

4. 流程控制

通过条件判断（if/else、switch）和循环（for、while）控制代码执行流程。

// if / else if / else
let score = 85;
if (score >= 90) {
  console.log("优秀");
} else if (score >= 70) {
  console.log("良好");   // ← 输出这里
} else {
  console.log("继续努力");
}

// switch
let day = 1;
switch (day) {
  case 1: console.log("星期一"); break;
  case 6: console.log("星期六"); break;
  default: console.log("其他");
}

// for 循环
for (let i = 1; i <= 5; i++) {
  console.log(`第 ${i} 次循环`);
}

// while 循环
let n = 3;
while (n > 0) {
  console.log(`倒计时：${n}`);
  n--;
}

5. 函数

函数是可复用的代码块，JavaScript 支持多种函数写法。

// 函数声明（有提升）
function add(x, y) {
  return x + y;
}
console.log(add(3, 5));   // 8

// 函数表达式
const multiply = function(x, y) {
  return x * y;
};

// 箭头函数（ES6，更简洁）
const square = x => x * x;
console.log(square(4));   // 16

// 带默认参数的函数
function greet(name = "同学") {
  return `你好，${name}！欢迎学习 JavaScript。`;
}
console.log(greet());           // 你好，同学！
console.log(greet("王五"));     // 你好，王五！

// 递归函数：计算阶乘
function factorial(n) {
  if (n <= 1) return 1;
  return n * factorial(n - 1);
}
console.log(factorial(5));   // 120
